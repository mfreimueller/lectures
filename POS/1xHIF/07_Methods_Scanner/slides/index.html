Scanner - Benutzereingabe

import java.util.Scanner;

public class Eingabe {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Name: ");
        String name = scanner.nextLine();

        System.out.print("Alter: ");
        int alter = scanner.nextInt();

        System.out.println("Hallo " + name + ", " + alter + " Jahre.");
    }
}

Methoden - Definition

// Methode ohne Rückgabe, ohne Parameter
public static void sayHello() {
    System.out.println("Hallo!");
}

// Methode mit Parameter, ohne Rückgabe
public static void sayHelloTo(String name) {
    System.out.println("Hallo " + name + "!");
}

// Methode mit Parameter und Rückgabe
public static int add(int a, int b) {
    return a + b;
}

Methoden - Aufruf

public static void main(String[] args) {
    sayHello();                    // Hallo!
    sayHelloTo("Anna");           // Hallo Anna!
    int summe = add(5, 3);        // summe = 8
    System.out.println(summe);    // 8
}

public static void sayHello() {
    System.out.println("Hallo!");
}

public static void sayHelloTo(String name) {
    System.out.println("Hallo " + name + "!");
}

public static int add(int a, int b) {
    return a + b;
}

Parameter und Ruckgabe

  • Parameter: Werte, die die Methode bekommt
  • Ruckgabetyp: void = nichts, sonst Typ des Ruckgabewerts
  • return: beendet die Methode und gibt Wert zurück

Kata 1: int max(int a, int b)

public static int max(int a, int b) {
    if (a > b) {
        return a;
    } else {
        return b;
    }
}

public static void main(String[] args) {
    int grösser = max(15, 8);
    System.out.println(grösser);  // 15
}

Kata 2: boolean istPalindrom

public static boolean istPalindrom(String wort) {
    int länge = wort.length();
    for (int i = 0; i < länge / 2; i++) {
        if (wort.charAt(i) != wort.charAt(länge - 1 - i)) {
            return false;
        }
    }
    return true;
}

Dungeon Escape: Methoden

public static void zeigeStatus(String name, int hp) {
    System.out.println("=== " + name + " ===");
    System.out.println("HP: " + hp);

    System.out.println("Du bist im: " + raumName);

    return hp - schaden;
}
